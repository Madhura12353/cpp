#include<iostream>
#include<iomanip>
using namespace std;

int main()
{

	system("color E0");
	system("title print the WHOLE ASCII table or alphabets n character per line by Madhura");

	int i, choice = 0, min_number = 0, max_number = 0, limitPerLine, num = 1;
	char a;
	const int ASCII_MIN_NUMBER = 0,
		ASCII_MAX_NUMBER = 255,
		CAPITAL_MIN_NUMBER = 65,
		CAPITAL_MAX_NUMBER = 90,
		LOWER_MIN_NUMBER = 97,
		LOWER_MAX_NUMBER = 122,
		DIGITS_MIN_NUMBER = 48,
		DIGITS_MAX_NUMBER = 56,
		DOTTED_LINE = 124;

	while (choice != 5)
		//giving menu to user
	{
		cout
			<< "\n\nPlease enter your choice:\n"
			<< "\t 1] Print the whole ASCII table\n"
			<< "\t 2] Print the capital letters\n"
			<< "\t 3] Print the lower letters\n"
			<< "\t 4] Print the digits\n"
			<< "\t 5] Exit the program\n"
			<< "Choice : ";
		//asking user input
		cin >> choice; cin.ignore(80, '\n');
		//switch case to asign veriable 

		switch (choice)
		{
		case 1: choice; min_number = ASCII_MIN_NUMBER, max_number = ASCII_MAX_NUMBER; break;
		case 2: choice; min_number = CAPITAL_MIN_NUMBER, max_number = CAPITAL_MAX_NUMBER; break;
		case 3: choice; min_number = LOWER_MIN_NUMBER, max_number = LOWER_MAX_NUMBER; break;
		case 4: choice; min_number = DIGITS_MIN_NUMBER, max_number = DIGITS_MAX_NUMBER; break;
		default:cout << endl << (choice < 1 || choice>5 ? "\a    *****  Invalid choice " : "") << endl;
			continue;
		}

		cout << "\nHow many characters do you want to print per line?";
		cin >> limitPerLine; cin.ignore(80, '\n');
		//validating user input 
		while (limitPerLine > 4)
		{
			cout << "       Your number must be less than or equal to 4\n";
			cout << "       Try again:";
			cin >> limitPerLine; cin.ignore(80, '\n');
		}

		// Printing user requird  values using for loop

		for (i = min_number, num = 1; i <= max_number; i++, num++)
		{
			a = i;
			cout << setw(5) << right << i;
			//Printing ascape sequences in ASCII table
			if (i == 0 || i == 7 || i == 8 || i == 9 || i == 10 || i == 11 || i == 13)
			{
				cout << setw(7) << right;
				cout << (i == 0 ? "NULL" :
					i == 7 ? "\\a" :
					i == 8 ? "\\b" :
					i == 9 ? "\\t" :
					i == 10 ? "\\n" :
					i == 11 ? "\\v" :
					i == 13 ? "\\r" : "");
				
			}//Going to new line 
			else cout << setw(7) << right << a;
			// Here is the end of line
			if (num % limitPerLine == 0) {
				cout << "\n";
			}
			// No end of line
			else {
				cout << setw(4) << right << char(DOTTED_LINE) << char(DOTTED_LINE);
			}

		}
	}

	cout << "Thank you for using printing with us. Please hit ENTER to end the program";
	cin.get();

}
